            ADD CatName varchar(10);

            ALTER Column CatName varchar(10) NOT NULL

Folosirea Constrangerilor
    NOT NULL        (Constrange la valoare nenula)
    UNIQUE          (Marcheaza ca unic un anumit camp)
    PRIMARY KEY     (Asigneaza cheia primara unei tabele)         
    FOREIGN KEY     (Creeaza o legatura cu alta tabela)
    CHECK           (Limiteaza domeniul de valori)
    DEFAULT         (Seteaza o valoare implicita cand nu este specificat nimic)
